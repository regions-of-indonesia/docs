## Installation

```bash
npm install @regions-of-indonesia/solid-query
# or
yarn add @regions-of-indonesia/solid-query
# or
pnpm add @regions-of-indonesia/solid-query
```

## Usage

```typescript
import { create } from "@regions-of-indonesia/solid-query";

import client from "./path/to/client";

const {
  createProvinces,
  createProvince,
  createDistricts,
  createDistrict,
  createSubdistricts,
  createSubdistrict,
  createVillages,
  createVillage,
  createRegion,
  createSearch,
  createSearchProvinces,
  createSearchDistricts,
  createSearchSubdistricts,
  createSearchVillages,
} = create(client);
```
